108  }

110  {
111  hashTable->remove(bufDescTable[clockHand].file,bufDescTable[clockHand].pageNo);//从hash表中删除该页
112  bufDescTable[clockHand].Clear();//初始化该buffer帧
113  frame = clockHand;//通过frame参数返回帧号
114  }catch(HashNotFoundException e){

120  }
